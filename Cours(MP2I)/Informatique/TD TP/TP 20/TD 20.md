type op_binaire = Et | Ou | Implique

type formule =
  | Tauto (*⊤*)
  | Anti  (*⊥*)
  | X of int (* dans 1..n *)
  | Non of formule
  | Bin of op_binaire * formule * formule


(*****************************************************************)
(*            Valuation et tables de vérités                     *)
(*****************************************************************)

(* Pour manipuler des tables de vérité ou des valuations, il peut être
   utile de commencer par déterminer l'ensemble des variables. *)
  
let rec varmax f = match f with
  (* calcule le plus grand indice des variables dans Var(f), 0 si f n'a aucune variable *)
  | Tauto | Anti    -> 0
  | X i             -> i
  | Non f1          -> varmax f1
  | Bin (_, f1, f2) -> max (varmax f1) (varmax f2)

type valuation = bool array
let rec eval (v : valuation) f = match f with
  (* eval v f calcule [f]_v *)
  | Tauto  -> true
  | Anti -> false
  | X i -> assert (1 <= i && i < Array.length v); v.(i)
  | Non f1 -> not (eval v f1)
  | Bin (Et, f1, f2) -> eval v f1 && eval v f2
  | Bin (Ou,  f1, f2) -> eval v f1 || eval v f2
  | Bin (Implique, f1, f2) -> not (eval v f1) || eval v f2
 
let table_verite f =
  let vmax = varmax f in
  let v = Array.make (vmax + 1) false in
  let val_verite b = if b then "V " else "F " in
  let rec remplir_a_partir_de i =
    if i = vmax + 1 then (
      for x = 1 to vmax do print_string (val_verite v.(x)) done;
      print_endline (" --> " ^ (val_verite (eval v f)))
    ) 
    else (
      v.(i) <- false;
      remplir_a_partir_de (i+1); 
      v.(i) <- true;  
      remplir_a_partir_de (i+1)
    ) 
  in
  remplir_a_partir_de 1 

let rec substitution i g f =
  match f with 
  | Tauto | Anti -> f
  | X j -> if j = i then g else f
  | Non f1 -> Non (substitution i g f1)
  | Bin (o, f1, f2) -> Bin(o, substitution i g f1, substitution i g f2)


let implique f1 f2 =
  match f1 with
  | Tauto -> f2
  | Anti -> Tauto
  | _ -> Bin(Ou, Non (f1), f2)

let et f1 f2 =
  match (f1, f2) with 
  |(Anti,_) | (_,Anti) -> Anti
  |(Tauto,g) | (g,Tauto) -> g
  | _ -> Bin(Et, f1,f2)

let ou f1 f2 =
  match (f1,f2) with
  |(Tauto,_) | (_,Tauto) -> Tauto
  |(Anti,g) | (g,Anti) -> g
  |_ -> Bin(Ou, f1, f2)

let rec nnf f =
  match f with
  | X j -> f
  | Tauto | Anti  -> f
  | Non(Tauto) -> Anti
  | Non(Anti) -> Tauto
  | (Non(Bin(Et,f1,f2))) -> ou (nnf(Non f1)) (nnf(Non f2))
  | (Non(Bin(Implique,f1,f2))) -> et (nnf(f1)) (nnf(Non f2))
  | (Non(Bin(Ou,f1,f2))) -> et (nnf(Non f1))(nnf(Non f2))
  | Non(Non f1) -> nnf f1
  | Non (X _) -> f
  | Bin(Implique, f1,f2) -> implique (nnf f1) (nnf f2)
  | Bin(Ou, f1,f2) -> ou (nnf f1) (nnf f2)
  | Bin(Et, f1,f2) -> et (nnf f1) (nnf f2)

let rec quine_sat f =
  let g = nnf f in
    match g with
    | Tauto -> true
    | Anti -> false
    | _ -> 
    let v = varmax g in
    quine_sat(substitution v Tauto g) || quine_sat(substitution v Anti g)

let rec quine_modele f =
  let rec aux i w i_w =
    let g = nnf f in
    if g = Anti then None
    else 
      let v = varmax g in
      let w_etendu = (List.init(i_w - v - 1) (fun _ -> true)) @ w in
      if g = Tauto then Some w_etendu
      else
        match aux (substitution v Tauto g) (true::w_etendu) v with
        | Some u -> Some u
        | None -> aux (substitution v Anti g) (false::w_etendu) v
  in
  aux f [] (varmax f+1)
  
let quine_affiche_modeles f =
  let rec aux f w= 
    let g = nnf f in
    if g = Tauto then 
    let affiche_couple (i,b) =
      print_string "x";
      print_int i;
      print_string (" : "^ (if b then "V" else "F") ^"   ")
      in
      List.iter affiche_couple w;
      print_newline ()
    else if g <> Anti then
      let i = varmax g in
      aux (substitution i Tauto g) ((i,true)::w);
      aux (substitution i Anti g) ((i,false)::w)
  in
  aux f []

let string_of_binop = function
  | Et     -> "/\\"
  | Ou      -> "\\/"
  | Implique -> "->"

let rec string_of_formule f =
  match f with
  | Tauto -> "⊤"
  | Anti -> "⊥"
  | X i -> "x" ^ (string_of_int i)
  | Non f1 -> "(¬" ^ (string_of_formule f1) ^ ")"
  | Bin (op, f1, f2) -> "(" ^ (string_of_formule f1) ^ " " ^(string_of_binop op) ^ " " ^ (string_of_formule f2) ^ ")"

let print f = print_endline (string_of_formule f)
